## 단지번호붙이기

![image](https://user-images.githubusercontent.com/52685247/74249787-10dce300-4d2d-11ea-8f2d-ce57f7f5855b.png)



- 작성코드

```python
def dfs(row, col):
    global cnt
    arr[row][col] = 0
    visit[row][col] = True
    for i in range(4):
        x, y = row+dx[i], col+dy[i]
        if 0 <= x < N and 0 <= y < N:
            if arr[x][y] == 1 and not visit[x][y]:
                cnt += 1
                dfs(x, y)

    return cnt

N = int(input())
arr = [list(map(int, input())) for _ in range(N)]
# print(arr)
visit = [[False] * N for _ in range(N)]
# print(visit)
dx, dy = [-1, 1, 0, 0], [0, 0, -1, 1]
result = []

cnt = 1
for i in range(N):
    for j in range(N):
        if arr[i][j] == 1 and not visit[i][j]:
            dfs(i, j)
            result.append(cnt)
            cnt = 1

result.sort()
print(len(result))
for i in result:
    print(i)
```
