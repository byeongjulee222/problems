#### 2차배열에서 상하좌우 값 뽑아내는 기본 틀

```python
N = 10          # N x N 행렬
dx = [-1, +1, 0, 0] # 상 하 좌 우
dy = [0, 0, -1, +1]

for x in range(N):
    for y in range(N):
        # [x][y]
        # 4방향의 인접 위치 좌표를 생성
        for i in range(4):
            tx, ty = x + dx[i], y + dy[i]
            # 경계 체크
            if tx < 0 or tx == N or ty < 0 or ty == N:
                continue
```



#### 연습문제 1 (2차배열 절댓값)

```python
arr = [[9, 20, 2, 18, 11],
[19, 1, 25, 3, 21],
[8, 24, 10, 17, 7],
[15, 4, 16, 5, 6],
[12, 13, 22, 23, 14]]

N, M = len(arr), len(arr[0])  # N : 행, M : 열

dx = [-1, +1, 0, 0]
dy = [0, 0, -1, +1]


result = []
SUM = 0
for x in range(N):
    for y in range(M):
        SUM = 0
        for i in range(len(dx)):
            tx, ty = x + dx[i], y + dy[i]
            if tx < 0 or tx == N or ty < 0 or ty == N:
                continue
            else:
                SUM += abs(arr[x][y] - arr[tx][ty])
        result.append(SUM)

MAX = result[0]
for j in range(len(result)):
    if MAX < result[j]:
        MAX = result[j]
print(MAX)
```

```python
arr = [[ 1,  2,  4,  7, 11],
    [ 3,  5,  8, 12, 15],
    [ 6,  9, 13, 16, 18],
    [10, 14, 17, 19, 20]]


N, M = len(arr), len(arr[0])
for diag in range(0, N + M - 1):    # diag: 사선의 수
                                    # x, y: 시작 좌표
    x = 0 if diag < M else (diag - M + 1)
    y = diag if diag < M else M - 1

    while x < N and y >= 0:
        print('%2d ' % arr[x][y], end='')
        x += 1
        y -= 1  # 다음 사선으로 넘어갈 때 행 += 1, 열 -= 1
    print()


```

##### 부분집합_binary counting

```python
arr = [3, 6, 7, 1, 5, 4]

for i in range(1 << len(arr)):  # i: 집합의 비트 표현
    for j in range(len(arr)):
        if i & (1 << j):
            print(arr[j], end=", ")

    print()
```



##### 예제) 부분집합 내의 원소 합이 0인 부분집합을 모두 출력

```python
arr = [3, 6, -2, 7, -3, 1, -5, -1, 5, 4]
count = 0
for i in range(1 << len(arr)):      # 2**len(arr) 수만큼 반복 / 부분집합을 표현
    result = []                     # 반복할 때마다 result 리스트를 비운다.
    for j in range(len(arr)):       # 원소의 수만큼 비트를 비교함
        # arr[j]를 포함하는지
        if i & (1 << j):            # i의 j번째 비트가 1이면 j번째 원소를 츨력
            result.append(arr[j])
    if sum(result) == 0:
        print(result)
        count += 1
print(count)
```



##### 2진 검색

```python
arr = []
key = 123
lo, hi = 0, len(arr) - 1

def binarySearch(arr, key):
    lo, hi = 0, len(arr) - 1

    while lo <= hi:
        mid = (lo+hi) >> 1
        if arr[mid] == key:
            return mid
        if arr[mid] > key: 
            hi = mid - 1
        else:
            lo = mid + 1
    return -1


# 재귀 호출
def binarySearch(arr, lo, hi, key):
    if lo > hi: return False

    mid = (lo+hi) >> 1
    if arr[mid] == key:
        return True
    if arr[mid] > key: 
        return binarySearch(arr, lo, mid-1, key)
    else:
        return binarySearch(arr, mid+1, hi, key)
```



##### 셀렉션 알고리즘

```python
arr = [64, 25, 10, 22, 11]
N = len(arr)

# [0, n-1] 범위내에서 최소값의 위치를 찾는다.
minIdx = 0
for j in range(minIdx+1, N):
    if arr[minIdx] > arr[j]:
        minIdx = j
arr[0], arr[minIdx] = arr[minIdx], arr[0]

# [1, n-1] 범위내에서 최소값의 위치를 찾는다.
minIdx = 1
for j in range(minIdx+1, N):
    if arr[minIdx] > arr[j]:
        minIdx = j
arr[1], arr[minIdx] = arr[minIdx], arr[1]
print(arr)

# 범위가 마지막 두 개가 될 때까지 반복
for i in range(N-1):
    minIdx = i
    for j in range(minIdx+1, N):
        if arr[minIdx] > arr[j]:
            minIdx = j
        arr[i], arr[minIdx] = arr[minIdx], arr[i]
print(arr)
```
